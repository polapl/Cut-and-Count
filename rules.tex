\begin{enumerate}
\item $u \in X_{t_0}$ oraz $v \in X_{t_0}$

\text{\footnotesize{\(c(t, (X_{t_0}, X_{t_1}, X_{t_2}), M) = c_{w/o\_edge}\)}}

\item $u \in X_{t_1}$, $v \in X_{t_1}$ oraz  $\{u,v\} \in M$ (\emph{przypadek 1} na rys. \ref{introduce_edge})

\text{\footnotesize{\(c(t, (X_{t_0}, X_{t_1}, X_{t_2}), M) =  c_{w/o\_edge} \vee c(t', (X_{t_0} \cup \{u,v\}, X_{t_1} \setminus \{u,v\}, X_{t_2}), M \setminus \{u,v\})\)}}

\item $u \in X_{t_1}$, $v \in X_{t_1}$ oraz  $\{u,v\} \notin M$

\text{\footnotesize{\(c(t, (X_{t_0}, X_{t_1}, X_{t_2}), M) =  c_{w/o\_edge}\)}}

\item \label{case_3}$u \in X_{t_1}$ oraz $v \in X_{t_2}$ (\emph{przypadek 2} na rys. \ref{introduce_edge})
    
\hspace*{-0.5cm}\text{\footnotesize{\(c(t, (X_{t_0}, X_{t_1}, X_{t_2}), M) = c_{w/o\_edge} \vee \bigvee_{\substack{M': \\ \{v,v'\} \in M'}} c(t', (X_{t_0} \cup \{u\}, X_{t_1} \cup \{v\} \setminus \{u\}, X_{t_2} \setminus \{v\}), \underbrace{M \setminus \{u,v'\} \cup \{v,v'\}}_\text{M'})\)}}    

\item $u \in X_{t_2}$ oraz $v \in X_{t_1}$ (analogicznie jak w punkcie \ref{case_3})
\item $u \in X_{t_2}$ oraz $v \in X_{t_2}$ (\emph{przypadki: 3, 4} na rys. \ref{introduce_edge})

\hspace*{-0.5cm}\text{\footnotesize{\(c(t, (X_{t_0}, X_{t_1}, X_{t_2}), M) = c_{w/o\_edge} \vee \bigvee_{\substack{M': \\ \{u,v\} \notin M' \\ \{u,u'\} \in M' \\ \{v,v'\} \in M'}} c(t', (X_{t_0}, X_{t_1} \cup \{u,v\}, X_{t_2} \setminus \{u,v\}), \underbrace{M \setminus \{u',v'\} \cup \{u,u'\} \cup \{v,v'\}}_\text{M'})\)}}
\end{enumerate}
